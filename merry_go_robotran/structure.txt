merry_go_robotran/
│
├── data/
│   ├── params.py            # Physical constants: masses, geometry, inertia, spring
│   ├── topology.py          # parent[], joint_type[], var_type[] — tree definition
│   └── initial_cond.py      # q0, qd0, t_end
│
├── neri/
│   ├── body.py              # Body dataclass (m, I, d_z_ii, etc.)
│   ├── joint.py             # Joint classes: R, C2, Fixed — provides φ(q), R(q)
│   ├── state.py             # MBState dataclass: ω, α, β, R0i, F, L per body
│   ├── forward.py           # Forward pass: fills state.ω, state.α for all bodies
│   ├── backward.py          # Backward pass: fills state.F, state.L; projects Q
│   └── assembler.py         # Last trick: builds M(q) and c(q,qd)
│
├── simulation/
│   ├── driven_vars.py       # qc(t), qdc(t), qddc(t) — control laws
│   ├── forces.py            # Motor torque, wind force, spring-damper — returns Q, Fext
│   ├── integrator.py        # yd=f(y,t) function + solve_ivp call
│   └── main.py              # Entry point: load → init → integrate → save → plot
│
├── postprocess/
│   ├── internal_forces.py   # Extract F^i, L^i at critical sections from saved states
│   ├── dimensioning.py      # σ, ε, safety factor, critical section search
│   └── plots.py             # All matplotlib figures
│
├── results/                 # Saved .npz output arrays
│
└── validation/
    └── gonogo.py            # Reproduce Go/No-Go curve, compare with MOOC reference
